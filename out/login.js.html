import { GoogleLogin } from 'react-google-login';
import {useNavigate} from "react-router-dom";
const clientId = "701943938891-m4tuu3ab7rui5fu18kgesgdnvn5bplg3.apps.googleusercontent.com";

/**
 * This funct
 * @returns {JSX.Element}
 * @constructor
 */


function Login(){

    const navigate = useNavigate();
    const changeToLanding = () =>{
        navigate('/order');
        // navigate('./customer_pages/menuOrdering', {state: {drinkIds}});
    }
    const onSuccess = (res) => {
        console.log("login works :" + res.profileObj)
        changeToLanding()
    }
    const onFailure = (res) => {
        console.log("login fails :" + res)
    }

    return(
    <div id = "signInButton">
        <GoogleLogin
            clientId={clientId}
            buttonText={"Login"}
            onSuccess={onSuccess}
            onFailure={onFailure}
            cookiePolicy={'single-host-origin'}
            isSignedIn={true}
        />
    </div>
    )
}
export default Login;
